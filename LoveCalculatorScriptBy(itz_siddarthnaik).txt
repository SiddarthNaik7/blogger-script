   function calculateLove() {
  var name1 = document.getElementById("name1").value.trim().toLowerCase();
  var name2 = document.getElementById("name2").value.trim().toLowerCase();
  
  var loveScore = getLoveScore(name1, name2);
  displayLoveResult(loveScore);
}

function getLoveScore(name1, name2) {
  var combinedNames = name1 + name2;
  var loveScore = 0;

  for (var i = 0; i < combinedNames.length; i++) {
    loveScore += combinedNames.charCodeAt(i);
  }

  loveScore = loveScore % 101;
  return loveScore;
}

function displayLoveResult(loveScore) {
  var loveResultElement = document.getElementById("loveResult");

  if (loveScore >= 80) {
    loveResultElement.innerHTML = "Congratulations! Your love score is " + loveScore + "%. You are a perfect match!";
  } else if (loveScore >= 50) {
    loveResultElement.innerHTML = "Your love score is " + loveScore + "%. You have a good chance of success!";
  } else {
    loveResultElement.innerHTML = "Your love score is " + loveScore + "%. Sorry, it doesn't look good!";
  }
}

      
      
      var heartElement = document.getElementById("heart");
if (loveScore >= 80) {
  heartElement.classList.add("full");
} else {
  heartElement.classList.remove("full");
}
